public class Solution {

    private final int[] nums = new int[]{1, 2, 3, 4, 5, 6, 7, 8, 9};

    public static void main(String[] args) {
        int k = 9;
        int n = 45;
        System.out.println(new Solution().combinationSum3(k, n));
    }

    public List<List<Integer>> combinationSum3(int k, int n) {
        List<List<Integer>> res = new ArrayList<>();

        List<Integer> path = new ArrayList<>(k);
        Set<Integer> set = new HashSet<>();
        backtrack(nums, set, res, path, n, k, 0);
        return res;
    }

    private void backtrack(int[] nums, Set<Integer> set,
                           List<List<Integer>> res, List<Integer> path, int n, int k, int idx) {
        if (idx == k) {
            if (set.containsAll(path)) {
                return;
            }
            int sum = path.stream().mapToInt(o -> o).sum();
            if (sum == n) {
                res.add(new ArrayList<>(path));
            }
            return;
        }

        for (int i = idx; i < nums.length; i++) {
            if (nums[i] == 0) {
                continue;
            }
            path.add(nums[i]);
            nums[i] = 0;
            backtrack(nums, set, res, path, n, k, idx + 1);
            nums[i] = path.get(path.size() - 1);
            path.remove(path.size() - 1);
        }
    }


}
